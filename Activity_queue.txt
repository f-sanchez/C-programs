  #include <stdio.h>
  #include <stdlib.h>
  #define SIZE 5

    int queue[SIZE];
    int rear = -1; //aka tail
    int front = -1; //aka head

void enqueue(int item){
    if (front == -1)
        front++;
        
    if (rear == SIZE-1){
        printf("Queue overflow.\n");
        return;
    } else {
        queue[++rear] = item;
        printf("element added to queue: %d\n", item);
    }
}

int dequeue(){
    if ((rear && front) == -1){
        printf("Queue underflow.\n");
    } 
    // else {

    //         // rear--;
    //     }
    // }
    return queue[++front];
    //rear--;
}

void display(){
    int i;
    // printf("ajgka");
    if ((front && rear) == -1){
        printf("Queue empty.\n");
    } else {
        printf("Queue: ");
        for (i=front; i<=rear; i++){
            if (i == rear){
                printf("[%d] =>%d) ", i, queue[i]);
            } else {
                printf("([%d] =>%d, ", i, queue[i]);
            }
            
        }
    }
    printf("\n");
}
