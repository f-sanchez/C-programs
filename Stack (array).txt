# include<iostream>

using namespace std;

class Stack
{
    int top;
    public:
    int a[10];  //Maximum size of Stack
    Stack()
    {
        top = -1;
    }
    
    // declaring all the function
    void push(int x);
    int pop();
    int isEmpty();
    void print(Stack c);
};

// function to insert data into stack
void Stack::push(int x)
{
    if(top >= 10)
    {
        cout << "Stack Overflow \n";
    }
    else
    {
        a[++top] = x;
        cout << "Element Inserted " << x << "\n";
    }
}

// function to remove data from the top of the stack
int Stack::pop()
{
    if(top < 0)
    {
        cout << "Stack Underflow \n";
        return 0;
    }
    else
    {
        int d = a[top--];
        // cout << d;
        return d;
    }
}

// function to check if stack is empty
int Stack::isEmpty()
{
    if(top < 0)
        return 1;
    
    return 0;
}

void Stack::print(Stack c){
    cout << "Stack elements: \n";
    while (!c.isEmpty()){
        cout << c.pop() << " ";
    }
}

// main function
int main() {

    Stack s1;
    s1.push(3);
    s1.push(5);
    s1.pop();
    s1.push(6);
    s1.push(7);
 
    s1.print(s1);
    
    /*
        preform whatever operation you want on the stack
    */
}
